\section{Listing des fonctionnalités principales} 
\textbf{Fonctionnalités liées aux paramétrages de l'application} : 
\begin{itemize}
    \item configuration de la simulation (type de compartiment, régime, période de simulation) ;
    \item enregistrement de la configuration dans un sous-répertoire dédié ;
    \item chargement d'une configuration enregistrée.\\
\end{itemize}

\textbf{Fonctionnalités de lecture des fichiers} : 
\begin{itemize}
    \item lecture des fichiers binaires de type WATBAL\_MB, HYD\_Q, AQ\_H (permanents ou transitoires) issus des simulations ;
    \item lecture des fichiers de mesure (format du fichier à préciser dans la configuration) ;
    \item conversion vers une dataframe sim/obs pour chaque point de mesure. \\
\end{itemize}

\textbf{Fonctionnalités de spatialisation et d'analyse} : 
\begin{itemize}
    \item spatialisation des variables hydrologiques (charge hydraulique en nappe, ETP, ETR, recharge, ruissellement) ;
    \item calcul du coefficient de ruissellement au droit des stations de mesure ;
    \item tracé des bassins versants, topographie des points de mesure à partir des bassins unitaires et du réseau hydrographique associé ;
    \item représentation graphique des régimes et du bilan hydrologique pluriannuels. \\
\end{itemize}

\textbf{Fonctionnalités d'aide à la calibration} : 
\begin{itemize}
    \item calcul des critères statistiques au droit des points de mesure ;
    \item graphique de comparaison des variables hydrologiques observées et simulées (Q et H) ;
    \item modification des fichiers de paramétrisation depuis une sélection d'entités ou depuis la table attributaire (valable pour les champs de paramétrisation des fonctions de production et des fichiers aquifères saturés).\\
\end{itemize}
